\subsection{Was ist ein Token?}
\label{sec:what_is_token}

Ein Token ist die kleinste Einheit, die ein LLM verarbeitet. Anders als natürliche Sprache in Wörtern denkt, arbeitet ein Transformer auf einer \textbf{vordefinierten Vokabelliste} von Tokens. Ein Token kann sein:
\begin{itemize}[leftmargin=*]
  \item Ein ganzes Wort (z.\,B. \texttt{the} = 1 Token)
  \item Ein Wortteil (z.\,B. \texttt{un} + \texttt{believe} + \texttt{able})
  \item Ein einzelnes Zeichen bei unbekannten Wörtern
  \item Whitespace und Satzzeichen (jeweils eigene Tokens)
\end{itemize}

Die Umwandlung von Text in Tokens heisst \textbf{Tokenisierung}. Der Tokenizer ist fester Bestandteil jedes Modells --- er wird beim Training mitgelernt und ändert sich nicht mehr.

\begin{lstlisting}[language=Python, label=lst:tiktoken_basic, caption={Tiktoken Basic Usage}]
import tiktoken

encoding = tiktoken.get_encoding("cl100k_base")  # GPT-4o, GPT-4, GPT-3.5
tokens = encoding.encode("Hallo Welt")
print(tokens)        # [9587, 1928]
print(len(tokens))   # 2
print(encoding.decode(tokens))  # "Hallo Welt"
\end{lstlisting}

\textbf{BPE (Byte-Pair-Encoding)} ist der dominierende Algorithmus. Drei Schritte:
\begin{enumerate}[leftmargin=*]
  \item \textbf{Zerlegung}: Jedes Zeichen wird einzeln als Token behandelt
  \item \textbf{Merging}: Die häufigsten Paare werden iterativ zu einem Token zusammengefasst
  \item \textbf{Vokabular}: Nach festgelegter Anzahl Merges entsteht ein Vokabular (GPT-4o: ca. 100k Tokens)
\end{enumerate}

\textbf{SentencePiece} ist ein \textbf{sprachunabhängiger} Tokenizer, der direkt auf rohem Text arbeitet, ohne Whitespace-Pre-Tokenisierung. Wird von Anthropic und Google verwendet.

\subsubsection{Listing 1: Chat-Messages zählen (OpenAI Format) [Production-Ready]}

\begin{lstlisting}[language=Python, label=lst:count_chat_tokens, caption={Chat-Messages token counting (OpenAI format)}]
import tiktoken
from typing import List, Dict

def count_chat_tokens(messages: List[Dict[str, str]], model: str = "gpt-4o") $\to$ int:
    """
    Zählt Tokens für OpenAI Chat Completion Format.
    Basiert auf OpenAI Cookbook: counting_tokens.ipynb
    """
    encoding = tiktoken.encoding_for_model(model)
    tokens = 0
    
    for msg in messages:
        # Role + content overhead (OpenAI: 4 tokens per message)
        tokens += 4
        for key, value in msg.items():
            tokens += len(encoding.encode(value))
            if key == "name":  # name field vorhanden
                tokens += 1
    
    # Assistant primer
    tokens += 2
    return tokens


def count_chat_tokens_anthropic(messages: List[Dict[str, str]], model: str = "claude-3-5-sonnet") $\to$ int:
    """
    Anthropic nutzt eigenes Tokenizer-Format.
    Workaround: cl100k_base als Approximation (ca. 5-10% Abweichung).
    Für exakte Counts: anthropic-tokenizer package oder API call mit stream=false, max_tokens=1.
    """
    encoding = tiktoken.get_encoding("cl100k_base")
    tokens = 0
    for msg in messages:
        tokens += 4  # role + content overhead
        for key, value in msg.items():
            tokens += len(encoding.encode(value))
    tokens += 2
    return tokens


# Usage
messages = [
    {"role": "system", "content": "Du bist ein Support-Agent."},
    {"role": "user", "content": "Mein Login funktioniert nicht."},
    {"role": "assistant", "content": "Verstehe. Welche Fehlermeldung sehen Sie?"},
    {"role": "user", "content": "Error 401: Unauthorized"},
]
print(f"Tokens (gpt-4o): {count_chat_tokens(messages, 'gpt-4o')}")      # ~45
print(f"Tokens (claude): {count_chat_tokens_anthropic(messages)}")      # ~47 (approx)
\end{lstlisting}

\subsubsection{Listing 2: Cost Estimation vor API-Call [Production-Ready]}
\label{lst:estimate_cost_token}

\begin{lstlisting}[language=Python, label=lst:estimate_cost, caption={Cost Estimation vor API Call}]
from dataclasses import dataclass
from typing import Dict

@dataclass
class ModelPricing:
    input_per_m: float   # $ per 1M input tokens
    output_per_m: float  # $ per 1M output tokens
    cached_per_m: float = 0.0  # $ per 1M cached input tokens (optional)

# Preise siehe Anhang A (aktuelle Tabelle)
PRICING: Dict[str, ModelPricing] = {
    "gpt-4o":           ModelPricing(2.50, 10.00, 1.25),   # 50% discount on cached
    "gpt-4o-mini":      ModelPricing(0.15, 0.60, 0.075),
    "gpt-4o-mini-2024-07-18": ModelPricing(0.15, 0.60, 0.075),
    "claude-3-5-sonnet": ModelPricing(3.00, 15.00),
    "claude-3-5-haiku":  ModelPricing(0.25, 1.25),
    "gemini-2.0-flash":  ModelPricing(0.075, 0.30),
    "gemini-2.0-pro":    ModelPricing(1.25, 10.00),
}

def estimate_cost(
    input_tokens: int,
    output_tokens: int,
    model: str,
    cached_tokens: int = 0
) $\to$ float:
    """
    Schätzt Kosten vor dem API-Call.
    Returns: geschätzte Kosten in USD.
    """
    pricing = PRICING.get(model)
    if not pricing:
        raise ValueError(f"Unbekanntes Modell: {model}")
    
    input_cost = (input_tokens - cached_tokens) * pricing.input_per_m / 1_000_000
    cached_cost = cached_tokens * pricing.cached_per_m / 1_000_000
    output_cost = output_tokens * pricing.output_per_m / 1_000_000
    
    return input_cost + cached_cost + output_cost


# Usage
input_toks = 1500
output_toks = 500
model = "gpt-4o"

est = estimate_cost(input_toks, output_toks, model)
print(f"Geschätzte Kosten: ${est:.6f}")  # ~$0.00875

# Mit Caching (z.B. 1000 Tokens System-Prompt gecacht)
est_cached = estimate_cost(input_toks, output_toks, model, cached_tokens=1000)
print(f"Mit Caching: ${est_cached:.6f}")  # ~$0.00550 (37% Ersparnis)
\end{lstlisting}

\subsubsection{Listing 3: Streaming Token Counter [Production-Ready]}
\label{lst:streaming_counter_token}

\begin{lstlisting}[language=Python, label=lst:streaming_counter, caption={Streaming Token Counter mit Budget-Enforcement}]
from dataclasses import dataclass
from enum import Enum
import tiktoken

class BudgetDecision(Enum):
    CONTINUE = "continue"
    WARN = "warn"      # 80% threshold reached
    STOP = "stop"      # Budget exceeded

@dataclass
class StreamingTokenCounter:
    model: str
    budget: int
    warning_threshold: float = 0.8
    
    def __post_init__(self):
        self.encoding = tiktoken.encoding_for_model(self.model)
        self.count = 0
        self.warning_triggered = False
    
    def add_chunk(self, chunk: str) $\to$ BudgetDecision:
        """
        Fügt Chunk hinzu, prüft Budget.
        Returns: Decision enum für Stream-Handling.
        """
        self.count += len(self.encoding.encode(chunk))
        
        if self.count > self.budget:
            return BudgetDecision.STOP
        elif self.count > self.budget * self.warning_threshold and not self.warning_triggered:
            self.warning_triggered = True
            return BudgetDecision.WARN
        return BudgetDecision.CONTINUE
    
    def reset(self):
        self.count = 0
        self.warning_triggered = False


# Usage in Streaming Loop
async def stream_with_budget(client, messages, model: str, budget: int):
    counter = StreamingTokenCounter(model, budget)
    response_chunks = []
    
    stream = await client.chat.completions.create(
        model=model,
        messages=messages,
        max_tokens=budget,  # Hard limit server-side
        stream=True
    )
    
    async for chunk in stream:
        if chunk.choices[0].delta.content:
            content = chunk.choices[0].delta.content
            decision = counter.add_chunk(content)
            
            if decision == BudgetDecision.STOP:
                # Stream abbrechen, partial response returnen
                yield {"content": "".join(response_chunks), "truncated": True, "tokens_used": counter.count}
                break
            elif decision == BudgetDecision.WARN:
                yield {"content": content, "warning": "80% budget reached"}
            else:
                yield {"content": content}
            
            response_chunks.append(content)
    
    # Final cost tracking
    final_cost = estimate_cost(
        input_tokens=count_chat_tokens(messages, model),
        output_tokens=counter.count,
        model=model
    )
    yield {"cost_usd": final_cost, "total_output_tokens": counter.count}
\end{lstlisting}

\subsection{Das Context Window}
\label{sec:context_window}

Das Context Window (Kontextfenster) ist die maximale Menge an Tokens, die ein Modell in einem Durchlauf verarbeiten kann. Es umfasst:
\begin{itemize}[leftmargin=*]
  \item \textbf{System-Prompt}: Verhaltensdefinition, Regeln, Persona
  \item \textbf{User-Prompt}: Die aktuelle Anfrage
  \item \textbf{Chat-History}: Vorherige Nachrichten im Gespräch
  \item \textbf{RAG-Kontext}: Dokumente, die zur Beantwortung relevant sind
  \item \textbf{Output}: Die Antwort des Modells (auch Teil des Fensters)
\end{itemize}

\subsection{Input vs. Output Tokens --- Asymmetrie}
\label{sec:input_output_asymmetry}

APIs berechnen Input- und Output-Tokens getrennt:

\begin{itemize}[leftmargin=*]
  \item \textbf{Input-Tokens}: Was du an das Modell schickst (Prompt + History + RAG). Typisch 70--90\,\% der Gesamtkosten bei RAG-Systemen.
  \item \textbf{Output-Tokens}: Was das Modell generiert. Output-Tokens sind rechenintensiver (autoregressive Generation).
\end{itemize}

\textbf{Output-Token-Multiplikator pro Provider (mit Quellen):}
\begin{itemize}[leftmargin=*]
  \item OpenAI: $\sim$4,0$\times$ (Input \$2,50 $$\to$$ Output \$10,00 / 1M)
  \item Anthropic: $\sim$5,0$\times$ (Input \$3,00 $$\to$$ Output \$15,00 / 1M)
  \item Google: $\sim$3,75$\times$ (Input \$0,35 $$\to$$ Output \$1,31 / 1M)
\end{itemize}

\subsubsection{Output-Control-Patterns [Production-Ready]}
\label{sec:output_control}

\begin{enumerate}[leftmargin=*]
  \item \textbf{\texttt{max\_tokens} als harte Grenze}: Immer setzen, auch bei Streaming. Serverseitig erzwungen.
  \item \textbf{\texttt{stop} Sequenzen}: Für strukturierte Ausgabe (z.\,B. \texttt{stop=["}\texttt{\textbackslash n}\texttt{}\texttt{}\texttt{"]} für Single-Line).
  \item \textbf{JSON Mode / Structured Outputs}: Reduziert Verbosity drastisch. OpenAI: \texttt{response\_format=\{ "type": "json\_object" \}}, Structured Outputs mit Pydantic Schema.
  \item \textbf{Streaming mit Budget-Check pro Chunk}: Siehe Listing \ref{lst:streaming_counter_token}.
\end{enumerate}

\begin{lstlisting}[language=Python, label=lst:structured_output, caption={Structured Output mit Pydantic (Output-Token Reduktion)}]
from pydantic import BaseModel
from openai import OpenAI

class TicketClassification(BaseModel):
    category: str  # "billing" | "technical" | "account" | "other"
    priority: int  # 1-5
    sentiment: str  # "positive" | "neutral" | "negative"
    requires_escalation: bool

client = OpenAI()

response = client.beta.chat.completions.parse(
    model="gpt-4o-2024-08-06",
    messages=[
        {"role": "system", "content": "Klassifiziere das Support-Ticket."},
        {"role": "user", "content": "Meine Rechnung ist falsch, ich wurde doppelt belastet!"},
    ],
    response_format=TicketClassification,
    max_tokens=100,  # Structured Output ist kompakter
)

ticket = response.choices[0].message.parsed
print(ticket.category)       # "billing"
print(ticket.priority)       # 4
print(ticket.requires_escalation)  # True
\end{lstlisting}

\praxishinweis{
Structured Outputs sparen 30--50\,\% Output-Tokens vs. freier Text, da das Modell kein "Erklärungs-Overhead" generiert. Zusätzlicher Vorteil: Kein Parsing-Fehler-Risiko.

\subsection{Lost in the Middle (Liu et al. 2024)}
\label{sec:lost_in_middle}

\merke{
Zitat: ``Retrieval accuracy at 50\% context position is 30--50\% lower than at 10\% or 90\% (n=20 models, 10 datasets).'' --- Liu et al. 2024
}

Modelle schenken dem Anfang und Ende eines langen Prompts mehr Aufmerksamkeit als der Mitte.

\begin{verbatim}
Retrieval-Genauigkeit über die Kontext-Position:

\subsection{Code: Conversation Budget Guard [Production-Ready]}
\label{sec:conv_budget_guard}

\begin{lstlisting}[language=Python, label=lst:conv_budget_guard, caption={Conversation Budget Guard (SupportPilot Pattern)}]
from dataclasses import dataclass, field
from collections import defaultdict
from enum import Enum
from typing import Dict

class BudgetDecision(Enum):
    OK = "ok"
    WARN = "warn"
    REJECT = "reject"

@dataclass
class ConversationBudgetGuard:
    """
    Enforces token budget per conversation.
    Thread-safe für async Nutzung mit asyncio.Lock.
    """
    max_tokens: int = 8000
    warning_threshold: float = 0.8
    
    # In-Memory für Demo; Production: Redis (siehe unten)
    usage: Dict[str, int] = field(default_factory=lambda: defaultdict(int))
    
    def check(self, conv_id: str, estimated_tokens: int) $\to$ BudgetDecision:
        current = self.usage[conv_id]
        projected = current + estimated_tokens
        
        if projected > self.max_tokens:
            return BudgetDecision.REJECT
        elif projected > self.max_tokens * self.warning_threshold:
            return BudgetDecision.WARN
        return BudgetDecision.OK
    
    def record(self, conv_id: str, actual_tokens: int):
        """Tatsächliche Token-Nutzung nach API-Call buchen."""
        self.usage[conv_id] += actual_tokens
    
    def reset(self, conv_id: str):
        """Conversation beendet / archiviert."""
        self.usage.pop(conv_id, None)
    
    def get_remaining(self, conv_id: str) $\to$ int:
        return max(0, self.max_tokens - self.usage[conv_id])


# Usage
guard = ConversationBudgetGuard(max_tokens=8000, warning_threshold=0.8)

conv_id = "conv_123"
estimated = count_chat_tokens(messages, "gpt-4o") + 500  # + output buffer

decision = guard.check(conv_id, estimated)
if decision == BudgetDecision.REJECT:
    # Compression triggern oder User benachrichtigen
    compressed = compress_history(messages, target_tokens=4000)
    estimated = count_chat_tokens(compressed, "gpt-4o") + 500
    decision = guard.check(conv_id, estimated)
    
if decision == BudgetDecision.WARN:
    logger.warning(f"Conversation {conv_id} at 80% budget")

# ... API Call ...

guard.record(conv_id, actual_tokens_used)
\end{lstlisting}

\subsection{Tenant-Level Daily Budget (Multi-Tenancy) [Production-Ready]}
\label{sec:tenant_budget}

\begin{lstlisting}[language=Python, label=lst:tenant_budget, caption={Tenant Budget Tracker mit Redis}]
import redis.asyncio as redis
from datetime import date
from decimal import Decimal

class TenantBudgetTracker:
    """
    Tägliches Kostenbudget pro Tenant.
    Nutzt Redis INCRBYFLOAT für atomare Counter.
    """
    DAILY_LIMIT_USD: Decimal = Decimal("50.00")  # Config per tier
    
    def __init__(self, redis_client: redis.Redis):
        self.redis = redis_client
    
    def _key(self, tenant_id: str) $\to$ str:
        return f"tenant:{tenant_id}:daily_cost:{date.today().isoformat()}"
    
    async def check_and_reserve(self, tenant_id: str, estimated_cost: Decimal) $\to$ bool:
        """
        Prüft Budget und reserviert geschätzte Kosten atomar.
        Returns: True wenn Budget OK, False wenn Limit erreicht.
        """
        key = self._key(tenant_id)
        current = await self.redis.get(key)
        current = Decimal(current) if current else Decimal("0")
        
        if current + estimated_cost > self.DAILY_LIMIT_USD:
            return False
        
        # Atomare Reservierung
        await self.redis.incrbyfloat(key, float(estimated_cost))
        # TTL auf 48h für Cleanup
        await self.redis.expire(key, 172800)
        return True
    
    async def get_usage(self, tenant_id: str) $\to$ Decimal:
        key = self._key(tenant_id)
        val = await self.redis.get(key)
        return Decimal(val) if val else Decimal("0")
    
    async def get_usage_pct(self, tenant_id: str) $\to$ float:
        usage = await self.get_usage(tenant_id)
        return float(usage / self.DAILY_LIMIT_USD * 100)


# Usage in Request Pipeline
async def handle_request(tenant_id: str, messages: list, model: str):
    input_tokens = count_chat_tokens(messages, model)
    estimated_output = min(2000, 8000 - input_tokens)  # konservativ
    estimated_cost = estimate_cost(input_tokens, estimated_output, model)
    
    tracker = TenantBudgetTracker(redis_client)
    if not await tracker.check_and_reserve(tenant_id, Decimal(str(estimated_cost))):
        raise BudgetExceededError(f"Tenant {tenant_id} daily limit reached")
    
    # ... API Call ...
    # Nach Call: Actual Cost buchen (Delta-Korrektur)
\end{lstlisting}

\section{Best Practices}
\label{sec:best_practices}

\begin{enumerate}[leftmargin=*]
  \item \textbf{Count Before Call} --- Immer \texttt{tiktoken} zählen vor API Request
  \item \textbf{Hard Limits Everywhere} --- Request, Conversation, Tenant, Day
  \item \textbf{Stream with Budget} --- Chunk-by-Chunk Counting, Early Stop
  \item \textbf{Compress Progressively} --- Sliding $$\to$$ Summary $$\to$$ Semantic (nicht alles auf einmal)
  \item \textbf{Cache with Tenant Isolation} --- Key = \texttt{tenant:prompt\_hash}
  \item \textbf{Route by Complexity} --- Simple $$\to$$ Cheap Model, Complex $$\to$$ Strong Model
  \item \textbf{Structure Output} --- JSON Mode / Structured Outputs spart Output Tokens
  \item \textbf{Log Structured Cost Data} --- model, in/out/cached tokens, cost, latency, tenant, feature
  \item \textbf{Alert on Anomalies} --- >2$\times$ p99 Cost, >10$\times$ Token Spike, Budget 80\%
  \item \textbf{Redact PII Before API} --- Nie rohe User-Daten in Context ohne Scan
\end{enumerate}
